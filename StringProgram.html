01. Reverse a word

package String01;

public class _01_Reverse_word {
    public static void main(String[] args) {
        String s = "java";
        String rev = "";
        for (int i = s.length() - 1; i >= 0; i--) {
            rev += s.charAt(i);
        }
        System.out.println(rev);
    }
}

02. Check if a word is a palindrome

03. Check if a word is a palindrome without using equals method

    package String01;
    
    public class _03_Palindrome_without_equals_method {
    
        static boolean pal(String s) {
            int si = 0;
            int ei = s.length() - 1;
    
            while (si < ei) {
                if (s.charAt(si) != s.charAt(ei)) {
                    return false;
                }
                si++;
                ei--;
            }
            return true;
        }
    
        public static void main(String[] args) {
    
            String s = "tenet";
    
            s = s.toLowerCase();
    
            if (pal(s)) {
                System.out.println("palindrome");
            } else {
                System.out.println("Not a Palindrome");
            }
        }
    
    }
    
    // Output: palindrome

04. Demonstrate replace() and replaceAll() methods

                System.out.println("palindrome :" + str.toLowerCase());
            else
                System.out.println("not a palindrome");
        }
    }

08. Count the number of characters in each word of a sentence.

            String s = "hi how are u";

            }

11. Check if a string is a palindrome (ignoring special characters)

    package String01;
    
    public class _11_palindrome {
    
        static boolean pal(String s) {
            int si = 0;
            int ei = s.length() - 1;
            while (si < ei) {
                if (s.charAt(si) != s.charAt(ei)) {
                    return false;
                }
                si++;
                ei--;
            }
            return true;
        }
    
        public static void main(String[] args) {

            String s1 = "@123LevE#13l#!";
    
            String s2 = s1.replaceAll("[^A-za-z]", "");
    
            String s3 = s2.toLowerCase();
    
            boolean res = pal(s3);
    
            if (res)
                System.out.println(s3 + " is palindrome");
            else
                System.out.println(s3 + " is not a palindrome");
    
        }
    
    }
    // Output: 
    // level is palindrome

12. Convert the case of each letter in a string (uppercase to lowercase and vice versa)

    package String01;
    
    public class _12_JaVa {
    
        public static void main(String[] args) {
    
            String s = "JaVa";
    
            for (int i = 0; i < s.length(); i++) {
                char ch = s.charAt(i);
    
                if (ch >= 'A' && ch <= 'Z') {
                    System.out.print((char) (ch + 32));
                } else if (ch >= 'a' && ch <= 'z') {
                    System.out.print((char) (ch - 32));
                }
    
            }
    
        }
    
    }
    // Output: 
    // jAvA

13. Find the frequency of each character in a string

    package String01;
    
    public class _13_Frequency_of_String {
    
        public static void main(String[] args) {
            String s = "anagram";
    
            int[] arr = new int[128];
    
            for (int i = 0; i < s.length(); i++) {
                arr[s.charAt(i)]++;
            }
    
            for (int i = 0; i < arr.length; i++) {
                if (arr[i] > 0) { 
                    System.out.println((char) i + " " + arr[i]);
                }
            }
    
        }
    }
    
    // Output:
    // a 3
    // n 1
    // g 1
    // r 1
    // m 1

14. Find the frequency of each character in a character array

    package String01;
    
    public class _14_Character_Frequency_in_char_Array {
    
        public static void main(String[] args) {
    
            char[] ch1 = { 'P', 'r', 'O', 'g', 'R', 'A', 'M', 'M', 'I', 'n', 'G' };
    
            int[] arr = new int[128];
    
            // char to String
    
            String s = new String(ch1);
    
            for (int i = 0; i < s.length(); i++) {
                char ch = s.charAt(i);
                arr[ch]++; // store
            }
    
            for (int i = 0; i < arr.length; i++) {
                if (arr[i] != 0) {
                    System.out.println((char) i + "  " + arr[i]);
                }
            }
    
        }
    }
    
    // Output:
    // P 1
    // r 1
    // O 1
    // g 1
    // R 2
    // A 1
    // M 2
    // I 1
    // n 1
    // G 1
